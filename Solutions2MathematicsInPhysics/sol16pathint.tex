\end{eqnarray*}

Since $Y=OX$, $D=OB$,
\begin{eqnarray*}
	\int_{-\infty}^\infty \exp\left( -i Y^T S Y +  i D^T Y \right)\prod_{i=1}^N \mathrm dy_i &=& \sqrt{\frac{\pi^N}{\det(S)}} \exp\left( \frac i 4 D^T S^{-1} D \right) 
\end{eqnarray*}
